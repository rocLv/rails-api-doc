def perform(task, *)
  require_rake

  ARGV.unshift(task) # Prepend the task, so Rake knows how to run it.

  Rake.application.standard_exception_handling do
    Rake.application.init("rails")
    Rake.application.load_rakefile
    Rake.application.top_level
  end
end
