def initialize(message = nil, original_exception = nil)
  if message
    ActiveSupport::Deprecation.warn("Passing #message is deprecated and has no effect. "                                            "#{self.class} will automatically capture the message "                                            "of the original exception.", caller)
  end

  if original_exception
    ActiveSupport::Deprecation.warn("Passing #original_exception is deprecated and has no effect. "                                            "Exceptions will automatically capture the original exception.", caller)
  end

  super($!.message)
end
